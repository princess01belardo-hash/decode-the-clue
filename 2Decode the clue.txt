    function ensureAudio() {
      if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
      if (audioCtx.state === "suspended") audioCtx.resume();
      return audioCtx;
    }

    function tone(freq, duration = 0.12, type = "sine", volume = 0.08) {
      const ctx = ensureAudio();
      const osc = ctx.createOscillator();
      const gain = ctx.createGain();
      osc.type = type;
      osc.frequency.value = freq;
      gain.gain.setValueAtTime(volume, ctx.currentTime);
      gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + duration);
      osc.connect(gain).connect(ctx.destination);
      osc.start();
      osc.stop(ctx.currentTime + duration);
    }

      tone(180, 0.18, "sawtooth", 0.055);
      setTimeout(() => tone(130, 0.16, "sawtooth", 0.045), 130);
    }

    function warnBeep() { tone(950, 0.065, "square", 0.04); }

    function drumrollSound() {
      let count = 0;
      const roll = setInterval(() => {
        tone(90 + Math.random() * 80, 0.045, "square", 0.035);
        count++;
        if (count > 20) clearInterval(roll);
      }, 55);
    }

    function speak(lines) {
      if (!window.speechSynthesis) return;
      const text = Array.isArray(lines) ? lines[Math.floor(Math.random() * lines.length)] : lines;
      const utterance = new SpeechSynthesisUtterance(text);
      utterance.rate = 0.95;
      utterance.pitch = 1.12;
      utterance.volume = 0.9;
      window.speechSynthesis.cancel();
      window.speechSynthesis.speak(utterance);
    }

    function toggleMusic() {
      const ctx = ensureAudio();
      if (musicOn) {
        musicNodes.forEach(node => { try { node.stop(); } catch (e) {} });
        musicNodes = [];
        musicOn = false;
        els.musicBtn.textContent = "♫ Mission Music: OFF";
        return;
      }
      musicOn = true;
      els.musicBtn.textContent = "♫ Mission Music: ON";
      const master = ctx.createGain();
      master.gain.value = 0.025;
      master.connect(ctx.destination);
      const bass = ctx.createOscillator();
      bass.type = "square";
      bass.frequency.value = 55;
      bass.connect(master);
      bass.start();
      musicNodes.push(bass);
      const pulse = setInterval(() => {
        if (!musicOn) { clearInterval(pulse); return; }
        [110, 82, 146, 98].forEach((f, i) => setTimeout(() => tone(f, 0.06, "square", 0.018), i * 130));
      }, 650);
      musicNodes.push({ stop: () => clearInterval(pulse) });
    }

    function createCodeRain() {
      const chars = "1010◇△CODEMISSIONAGENT";
      for (let i = 0; i < 46; i++) {
        const col = document.createElement("div");
        col.className = "rain-column";
        col.style.left = `${Math.random() * 100}%`;
        col.style.animationDuration = `${5 + Math.random() * 7}s`;
        col.style.animationDelay = `${Math.random() * -8}s`;
        col.textContent = Array.from({ length: 18 }, () => chars[Math.floor(Math.random() * chars.length)]).join("\n");
        els.codeRain.appendChild(col);
      }
    }

    function loadQuestion() {
      const q = QUESTIONS[currentIndex];
      answered = false;
      els.itemLabel.textContent = `ITEM ${currentIndex + 1}`;
      els.icons.textContent = q.icons;
      els.clueText.innerHTML = q.clue;
      els.questionType.textContent = q.type;
      els.decodeZone.className = "decode-zone";
      els.decodePlaceholder.innerHTML = "⬇ Drag answer here to decode the clue ⬇";
      els.nextBtn.disabled = true;
      renderChoices(q.choices);
      syncTeacherFields();
      startTimer();
      updateStats();
    }

    function renderChoices(choices) {
      els.choices.innerHTML = "";
      choices.forEach(choice => {
        const tag = document.createElement("div");
        tag.className = "answer-tag";
        tag.draggable = true;
        tag.textContent = choice;
        tag.dataset.answer = choice;
        tag.addEventListener("dragstart", e => e.dataTransfer.setData("text/plain", choice));
        tag.addEventListener("click", () => checkAnswer(choice, tag));
        els.choices.appendChild(tag);
      });
    }

    function startTimer() {
      clearInterval(timerInterval);
      timeLeft = timerDuration;
      questionStart = Date.now();
      lastWarnSecond = null;
      updateTimerDisplay();
      timerInterval = setInterval(() => {
        timeLeft--;
        updateTimerDisplay();
        if (timeLeft <= 10 && timeLeft > 0 && lastWarnSecond !== timeLeft) {
          lastWarnSecond = timeLeft;
          warnBeep();
        }
        if (timeLeft <= 0) {
          clearInterval(timerInterval);
          els.timerDisplay.textContent = "0";
          els.decodeZone.classList.add("wrong");
          speak("Try again.");
          wrongSound();
        }
      }, 1000);
    }

    function updateTimerDisplay() {
      els.timerDisplay.textContent = timeLeft;
      els.timerBox.classList.toggle("low", timeLeft <= 10);
    }

    function pauseTimer() { clearInterval(timerInterval); }

    function checkAnswer(choice, tagEl) {
      if (answered) return;
      attempts++;
      const q = QUESTIONS[currentIndex];
      if (choice === q.answer) {
        answered = true;
        pauseTimer();
        correctCount++;
        score += 10;
        const elapsed = Math.max(0.1, (Date.now() - questionStart) / 1000).toFixed(1);
        leaderboard.push({ agent: activeAgent, time: Number(elapsed), item: currentIndex + 1 });
        leaderboard.sort((a, b) => a.time - b.time);
        leaderboard = leaderboard.slice(0, 5);
        els.decodeZone.classList.add("correct");
        els.decodePlaceholder.innerHTML = `<span class="answer-tag correct-choice">${choice} ✅</span>`;
        if (tagEl) tagEl.classList.add("correct-choice");
        successSound();
        makeConfetti();
        greenFlash();
        showPopup();
        speak(["Excellent decoding!", "Mission accomplished!", "Access granted!"]);
        els.nextBtn.disabled = false;
      } else {
        els.decodeZone.classList.remove("wrong");
        void els.decodeZone.offsetWidth;
        els.decodeZone.classList.add("wrong");
        if (tagEl) {
          tagEl.classList.remove("wrong-choice");
          void tagEl.offsetWidth;
          tagEl.classList.add("wrong-choice");
          setTimeout(() => tagEl.classList.remove("wrong-choice"), 800);
        }
        wrongSound();
        speak("Try again.");
      }
      updateStats();
    }

    function nextQuestion() {
      if (currentIndex < QUESTIONS.length - 1) {
        currentIndex++;
        loadQuestion();
      } else {
        pauseTimer();
        els.successPopup.classList.add("active");
        els.successPopup.querySelector("h2").textContent = "MISSION COMPLETE";
        els.successPopup.querySelector("p").textContent = "All clues decoded!";
        speak("Mission accomplished!");
        setTimeout(() => els.successPopup.classList.remove("active"), 1800);
      }
    }

    function updateStats() {
      const acc = attempts ? Math.round((correctCount / attempts) * 100) : 0;
      els.scoreDisplay.textContent = score;
      els.accuracyDisplay.textContent = `${acc}%`;
      els.progressText.textContent = `${correctCount}/${QUESTIONS.length}`;
      els.progressFill.style.width = `${(correctCount / QUESTIONS.length) * 100}%`;
      renderLeaderboard();
    }

    function renderLeaderboard() {
      els.leaderboard.innerHTML = "";
      if (leaderboard.length === 0) {
        const li = document.createElement("li");
        li.innerHTML = "<span>No decoded clues yet</span><span>--</span><span>--</span>";
        els.leaderboard.appendChild(li);
        return;
      }
      leaderboard.forEach(row => {
        const li = document.createElement("li");
        li.innerHTML = `<span>Agent #${row.agent}</span><span>${row.time}s</span><span>Item ${row.item}</span>`;
        els.leaderboard.appendChild(li);
      });
    }

    function makeConfetti() {
      const colors = ["#7CFC00", "#3FA9F5", "#F5F7E8", "#FF8C42", "#FF3B4F"];
      for (let i = 0; i < 60; i++) {
        const piece = document.createElement("div");
        piece.className = "confetti-piece";
        piece.style.background = colors[Math.floor(Math.random() * colors.length)];
        piece.style.setProperty("--x", `${(Math.random() - 0.5) * 620}px`);
        piece.style.setProperty("--y", `${(Math.random() - 0.5) * 440}px`);
        piece.style.transform = `rotate(${Math.random() * 360}deg)`;
        document.body.appendChild(piece);
        setTimeout(() => piece.remove(), 1050);
      }
    }

    function greenFlash() {
      els.screenFlash.classList.remove("active");
      void els.screenFlash.offsetWidth;
      els.screenFlash.classList.add("active");
    }

    function showPopup() {
      els.successPopup.querySelector("h2").textContent = "ACCESS GRANTED";
      els.successPopup.querySelector("p").textContent = "Code Decoded!";
      els.successPopup.classList.add("active");
      setTimeout(() => els.successPopup.classList.remove("active"), 1300);
    }

    function pickAgent() {
      const max = Math.max(1, Number(els.studentCount.value) || 1);
      els.agentResult.classList.add("drumroll");
      drumrollSound();
      let ticks = 0;
      const roll = setInterval(() => {
        const temp = Math.floor(Math.random() * max) + 1;
        els.agentResult.textContent = `Agent #${temp}`;
        ticks++;
        if (ticks > 20) {
          clearInterval(roll);
          activeAgent = Math.floor(Math.random() * max) + 1;
          els.agentResult.textContent = `Agent #${activeAgent}`;
          els.agentResult.classList.remove("drumroll");
        }
      }, 60);
    }

    function restartMission(resetScores = false) {
      pauseTimer();
      currentIndex = 0;
      if (resetScores) {
        score = 0;
        attempts = 0;
        correctCount = 0;
        leaderboard = [];
      }
      loadQuestion();
    }

    function syncTeacherFields() {
      const q = QUESTIONS[currentIndex];
      els.timerInput.value = timerDuration;
      els.studentCount.value = els.teacherStudents.value;
      els.editClue.value = q.clue.replace(/<br>/g, "\n");
      els.editAnswer.value = q.answer;
      els.editChoices.value = q.choices.join(", ");
    }

    function saveCurrentQuestion() {
      const q = QUESTIONS[currentIndex];
      q.clue = els.editClue.value.replace(/\n/g, "<br>");
      q.answer = els.editAnswer.value.trim();
      q.choices = els.editChoices.value.split(",").map(x => x.trim()).filter(Boolean).slice(0, 3);
      if (!q.choices.includes(q.answer)) q.choices.unshift(q.answer);
      q.choices = q.choices.slice(0, 3);
      timerDuration = Math.max(5, Number(els.timerInput.value) || 30);
      els.teacherStudents.value = Math.max(1, Number(els.teacherStudents.value) || 1);
      els.studentCount.value = els.teacherStudents.value;
      loadQuestion();
    }

    els.decodeZone.addEventListener("dragover", e => {
      e.preventDefault();
      els.decodeZone.classList.add("drag-over");
    });

    els.decodeZone.addEventListener("dragleave", () => els.decodeZone.classList.remove("drag-over"));

    els.decodeZone.addEventListener("drop", e => {
      e.preventDefault();
      els.decodeZone.classList.remove("drag-over");
      const choice = e.dataTransfer.getData("text/plain");
      const tagEl = [...document.querySelectorAll(".answer-tag")].find(el => el.dataset.answer === choice);
      checkAnswer(choice, tagEl);
    });

    els.nextBtn.addEventListener("click", nextQuestion);
    els.restartBtn.addEventListener("click", () => restartMission(false));
    els.restartTeacherBtn.addEventListener("click", () => restartMission(false));
    els.resetScoresBtn.addEventListener("click", () => restartMission(true));
    els.saveQuestionBtn.addEventListener("click", saveCurrentQuestion);
    els.pickAgentBtn.addEventListener("click", pickAgent);
    els.musicBtn.addEventListener("click", toggleMusic);

    els.studentCount.addEventListener("change", () => {
      els.teacherStudents.value = els.studentCount.value;
    });

    els.teacherStudents.addEventListener("change", () => {
      els.studentCount.value = els.teacherStudents.value;
    });

    els.timerInput.addEventListener("change", () => {
      timerDuration = Math.max(5, Number(els.timerInput.value) || 30);
      startTimer();
    });

    createCodeRain();
    loadQuestion();
  </script>
